**#1 Aktualizacja systemu**

sudo apt update && sudo apt upgrade -y


**#2 Instalacja Apache (serwer WWW)**

sudo apt install -y apache2
sudo systemctl enable apache2
sudo systemctl start apache2

**#3 Instalacja PHP i wymaganych rozszerzeń**

sudo apt install -y php php-mysql php-curl php-mbstring php-xml php-bcmath php-zip php-gd php-cli php-common php-tokenizer php-fileinfo php-mysqlnd php-intl php-readline php-soap php-exif php-fpm php-opcache php-imap php-ldap php-sqlite3 libapache2-mod-php

php -v


**#4 Instalacja MariaDB (MySQL)**

sudo apt install -y mariadb-server
sudo systemctl enable mariadb
sudo systemctl start mariadb

sudo mysql_secure_installation


**#5 Konfiguracja bazy danych Snipe-IT**

sudo mysql -u root -p

CREATE DATABASE snipeit CHARACTER SET utf8 COLLATE utf8_unicode_ci;
CREATE USER 'snipeuser'@'localhost' IDENTIFIED BY 'ustaw_silne_haslo';
GRANT ALL PRIVILEGES ON snipeit.* TO 'snipeuser'@'localhost';
FLUSH PRIVILEGES;
EXIT;

**#6 Włączenie modułów Apache**
sudo a2enmod rewrite
sudo systemctl restart apache2


**#7 Pobranie Snipe-IT v8.3.2**
cd /var/www/
sudo git clone -b v8.3.2 https://github.com/grokability/snipe-it.git
sudo chown -R $USER:$USER snipe-it
cd snipe-it

**#8 Instalacja Composer**

cd ~
curl -sS https://getcomposer.org/installer | php
sudo mv composer.phar /usr/local/bin/composer


**#9 Instalacja zależności PHP**
cd /var/www/snipe-it
composer install --no-dev --prefer-source


**#10 Konfiguracja .env**
cp .env.example .env
nano .env

**Zmień ustawienia:**

APP_URL=ip serwera ubuntu
APP_TIMEZONE='Europe/Warsaw'

DB_CONNECTION=mysql
DB_HOST=127.0.0.1
DB_DATABASE=snipeit
DB_USERNAME=snipeuser
DB_PASSWORD=silne_haslo

**#11 Wygenerowanie klucza aplikacji**

php artisan key:generate


**#12 Uprawnienia**

sudo chown -R www-data:www-data /var/www/snipe-it
sudo chmod -R 755 /var/www/snipe-it/storage
sudo chmod -R 755 /var/www/snipe-it/bootstrap/cache


**#13 Konfiguracja Apache dla Snipe-IT**

sudo nano /etc/apache2/sites-available/snipeit.conf


**Wklej konfiguracje i zrób zmiany**


<VirtualHost *:80>
    ServerName IP Serwera Ubuntu
    DocumentRoot /var/www/snipe-it/public

    <Directory /var/www/snipe-it/public>
        AllowOverride All
        Require all granted
    </Directory>

    ErrorLog ${APACHE_LOG_DIR}/snipeit_error.log
    CustomLog ${APACHE_LOG_DIR}/snipeit_access.log combined
</VirtualHost>


**Aktywacja:**

sudo a2ensite snipeit.conf
sudo a2dissite 000-default.conf
sudo systemctl reload apache2


**#14 Inicjalizacja bazy danych**

php artisan migrate --seed
